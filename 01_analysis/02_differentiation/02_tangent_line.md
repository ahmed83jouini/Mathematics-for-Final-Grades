elements:
  - h3: "1. المفهوم الهندسي للعدد المشتق"
    id: "geo-interpretation"

  - parags:
      - rtl: "عندما يقترب $h$ من الصفر في نسبة التغير، فإن القاطع $(AM)$ يقترب من وضعية حدية يصبح فيها يمس المنحنى في النقطة $A$."
      - rtl: "من هنا نستنتج أن العدد المشتق $f'(x_0)$ هو <strong>معامل توجيه</strong> (أو ميل) المماس للمنحنى $(C_f)$ عند النقطة ذات الفاصلة $x_0$."

  - h3: "2. معادلة المماس"
    id: "tangent-equation"

  - parags:
      - rtl: "بما أن المماس مستقيم معامل توجيهه $f'(x_0)$ ويمر بالنقطة $A(x_0; f(x_0))$، فإن معادلته تعطى بالقانون التالي:"
      - center: "$y = f'(x_0)(x - x_0) + f(x_0)$"

  - NB:
      title: "تنبيه في الحساب:"
      type: "primary"
      NBs:
        - rtl: "لحساب معادلة المماس، نحتاج دائماً إلى 'مكونين': الفاصلة $x_0$ والدالة $f$. نقوم بحساب $f(x_0)$ أولاً، ثم نحسب المشتقة $f'(x_0)$."

  - h3: "3. حالات خاصة للمماس"
    id: "special-cases"

  - tableLines:
      dir: rtl
      lines:
        - - th: "الحالة"
          - th: "التفسير الهندسي"
          - th: "النتيجة الرياضية"
        - - th: "مماس أفقي"
          - td: "المماس يوازي محور الفواصل"
          - td: "$f'(x_0) = 0$"
        - - th: "نقطة زاوية"
          - td: "الاشتقاق من اليمين لا يساوي اليسار"
          - td: "وجود نصفي مماس"

  - NB:
      title: "معادلة المماس العمودي:"
      type: primary
      NBs:
        - rtl: "إذا وجدنا أن نهاية نسبة التغير هي $\\pm\\infty$، فإن المنحنى يقبل مماساً عمودياً (موازياً لمحور :"
        - center: "التراتيب) معادلته $x = x_0$."

  - h3: "4. مثال توضيحي"
    id: "tangent-example"

  - parags:
      - rtl: "لتكن الدالة:"
      - center: "$f(x) = x^2 + 1$."
      - rtl: "لنعيين معادلة المماس عند $x_0 = 1$:"

  - tableLines:
      dir: rtl
      lines:
        - - th: "1. حساب الصورة"
          - td: "$f(1) = 1^2 + 1 = 2$"
        - - th: "2. حساب العدد المشتق"
          - td: "وجدنا سابقاً أن $f'(1) = 2$"
        - - th: "3. التعويض في القانون"
          - td: "$y = 2(x - 1) + 2 \\implies y = 2x$"

  - NB:
      title: "خلاصة هندسية:"
      type: "success"
      NBs:
        - rtl: "المستقيم ذو المعادلة :"
        - center: "$y = 2x$"
        - rtl: هو أفضل تقريب خطي للدالة بجوار النقطة $1$."
