elements:
  - h3: "1. القيمة الحدية المحلية."
    id: "local-extrema"

  - parags:
      - rtl: "نقول أن الدالة $f$ تقبل قيمة حدية محلية عند $x_0$ إذا كانت 'أعلى' أو 'أدنى' نقطة في جوار صغير يحيط بـ $x_0$."
      - rtl: "شرط وجود قيمة حدية محلية هو انعدام المشتقة <strong>وتغيير إشارتها</strong> عند تلك النقطة."

  - NB:
      title: "المبرهنة الأساسية:"
      type: "success"
      NBs:
        - rtl: "إذا انعدمت المشتقة $f'$ عند $x_0$ وغيرت إشارتها، فإن $f(x_0)$ هي قيمة حدية محلية للدالة $f$."

  - h3: "2. الفرق بين القيمة العظمى والصغرى"
    id: "max-min-diff"

  - tableLines:
      dir: rtl
      lines:
        - - th: "الحالة"
          - th: "التغير في الإشارة"
          - th: "نوع القيمة الحدية"
        - - td: "من الموجب إلى السالب ($+ \\to -$)"
          - td: "الدالة تصعد ثم تنزل"
          - td: "قيمة حدية <strong>عظمى</strong> محلية"
        - - td: "من السالب إلى الموجب ($- \\to +$)"
          - td: "الدالة تنزل ثم تصعد"
          - td: "قيمة حدية <strong>صغرى</strong> محلية"

  - h3: "3. القيمة الحدية المطلقة."
    id: "absolute-extrema"

  - parags:
      - rtl: "هي أعلى (أو أدنى) قيمة تصل إليها الدالة على</strong>كامل مجال تعريفها</strong> $D_f$، وليست فقط في جوار النقطة."
      - rtl: "ملاحظة: كل قيمة حدية مطلقة هي بالضرورة محلية، ولكن العكس ليس صحيحاً دائماً."

  - NB:
      title: "تنبيه (انعدام بدون تغيير إشارة):"
      type: "danger"
      NBs:
        - rtl: "إذا انعدمت المشتقة عند $x_0$ <strong>ولم تغير</strong> إشارتها (مثلاً بقيت موجبة قبل وبعد $x_0$)، فلا توجد قيمة حدية؛ بل نكون أمام 'نقطة انعطاف' حيث يواصل المنحنى صعوده بعد توقف لحظي."

  - h3: "4. مثال من جدول التغيرات"
    id: "example-extrema"

  - parags:
      - rtl: "من خلال جدول التغيرات للدالة :"
      - center: "$f(x) = x^3 - 3x$"
      - rtl: "الذي أنجزناه سابقاً:"

  - tableLines:
      dir: rtl
      lines:
        - - td: "عند $x = -1$"
          - td: "المشتقة انعدمت وغيرت إشارتها من $+$ إلى $-$"
          - td: "$f(-1) = 2$ هي قيمة حدية عظمى محلية"
        - - td: "عند $x = 1$"
          - td: "المشتقة انعدمت وغيرت إشارتها من $-$ إلى $+$"
          - td: "$f(1) = -2$ هي قيمة حدية صغرى محلية"

  - notes:
      - rtl: "بيانياً، القيم الحدية هي النقاط التي يكون عندها المماس <strong>أفقياً</strong> (معامل توجيهه صفر)."
