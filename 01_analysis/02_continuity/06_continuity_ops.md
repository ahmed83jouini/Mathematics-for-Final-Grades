  - tableLines:

  - h3: "2. استمرارية الدالة المركبة (تركيب دوال)"

  - parags:
      - rtl: "هذه هي القاعدة الأهم في التمارين المعقدة. لكي تكون الدالة المركبة $g \\circ f$ مستمرة، يجب أن يتحقق تسلسل في 'التوصيل':"

  - tableLines:
      dir: rtl
      lines:
        - - th: "المرحلة"
          - th: "الشرط المطلوب"

        - - td: "2. الدالة الثانية ($g$)"
          - td: "يجب أن تكون $g$ مستمرة على مجال يحتوي على صور $f(I)$."
        - - td: "3. النتيجة"
          - td: "الدالة $g(f(x))$ مستمرة على المجال $I$."

  - h3: "3. مثال تطبيقي مشروح"
    id: "composite-example"

  - parags:
      - rtl: "دراسة استمرارية الدالة:"
      - center: "$h(x) = \\sqrt{x^2 + 1}$"
      - rtl: "على $\\mathbb{R}$."

  - tableLines:
      dir: rtl
      lines:
        - - th: "التفكيك"
          - th: "التبرير"
        - - td: "الدالة الداخلية: $u(x) = x^2 + 1$"
          - td: "مستمرة على $\\mathbb{R}$ لأنها كثير حدود، وقيمها دائماً موجبة ($u(x) \\ge 1$)."
        - - td: "الدالة الخارجة $v(x) = \\sqrt{x}$"
          - td: "مستمرة على $[0, +\\infty[$ لأنها دالة مرجعية."
        - - td: "التركيب $v \\circ u$"
          - td: "بما أن $u(\\mathbb{R}) \\subset [0, +\\infty[$، فإن $h$ مستمرة على $\\mathbb{R}$."
