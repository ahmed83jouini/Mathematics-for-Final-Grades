elements:
  - h3: "1. قوة الرتابة (La Monotonie)"
    id: "monotony-power"

  - parags:
      - rtl: "في المبحث السابق، قلنا أن الاستمرارية تضمن وجود حل واحد على الأقل، لكن الدالة قد 'تتذبذب' وتمر على القيمة $k$ عدة مرات. لكي نضمن وجود حل وحيد، يجب أن تكون الدالة <strong >رتيبة تماماً</strong> (متزايدة تماماً أو متناقصة تماماً)."

  - h3: "2. نص المبرهنة (الوجود والوحدانية)"
    id: "mvt-unique-text"

  - parags:
      - rtl: "إذا كانت $f$ دالة مستمرة ورتيبة تماماً على مجال $[a, b]$، فإنه من أجل كل عدد حقيقي $k$ محصور بين $f(a)$ و $f(b)$، توجد<strong> قيمة وحيدة</storng> $\\alpha$ محصورة بين $a$ و $b$ بحيث:"
      - rtl: "$$f(\\alpha) = k$$"

  - h3: "3. الشروط الثلاثة الذهبية"
    id: "golden-conditions"

  - tableLines:
      dir: rtl
      lines:
        - - th: "الشرط"
          - th: "الدور في البرهان"
        - - td: "1. الاستمرارية"
          - td: "تضمن عدم وجود فجوات (الطريق موصول)."
        - - td: "2. الرتابة تماماً"
          - td: "تضمن أن الدالة تسير في اتجاه واحد ولا تعود لتقطع القيمة $k$ ثانية."
        - - td: "3. الحصر ($k$ بين الصور)"
          - td: "يضمن أن الهدف يقع داخل نطاق المجال المدروس."

  - h3: "4. التفسير البياني"
    id: "graph-interpretation"

  - parags:
      - rtl: "هندسياً، الرتابة تعني أن منحنى الدالة يقطع المستقيم الأفقي ذو المعادلة $y=k$ في <strong>نقطة واحدة فقط</strong>."

  - tableLines:
      dir: rtl
      lines:
        - - th: "سلوك الدالة"
          - th: "عدد الحلول الممكنة"
        - - td: "مستمرة فقط"
          - td"حل واحد، اثنان، أو مئة (على الأقل حل)."
        - - td: "مستمرة + رتيبة تماماً"
          - td: "حل وحيد لا غير."

  - h3: "5. مثال تطبيقي (منهجية الإجابة)"
    id: "mvt-unique-ex"

  - parags:
      - rtl: "لتكن الدالة:"
      - center: "$f(x)=x^3+x-5$"
      - rtl: "بين أن المعادلة:"
      - center: "$f(x)=0$"
      - rtl: "تقبل حلاً وحيداً $\alpha$ في $[1, 2]$."

  - tableLines:
      dir: rtl
      lines:
        - - th: "المرحلة"
          - th: "التبرير الرياضي"
        - - th: "1. الاستمرارية"
          - td: "$f$ دالة كثير حدود فهي مستمرة على $[1, 2]$."
        - - th: "2. الرتابة"
          - td: "$f'(x)=3x^2+1$ (موجبة تماماً)، إذن $f$ متزايدة تماماً على $[1, 2]$."
        - - th: "3. الحصر"
          - td: "$f(1)=-3$ و $f(2)=5$. بما أن $0 \\in [-3, 5]$."
        - - th: "4. الخلاصة"
          - td: "حسب مبرهنة القيم المتوسطة، المعادلة $f(x)=0$ تقبل حلاً وحيداً $\alpha$ في $]1, 2[$."
